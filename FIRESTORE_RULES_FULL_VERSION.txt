rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    function isAdmin() {
      return isAuthenticated() && 
             request.auth.token.admin == true;
    }
    
    function hasVerifiedEmail() {
      return isAuthenticated() && 
             request.auth.token.email_verified == true;
    }
    
    function isValidPollData() {
      let data = request.resource.data;
      return data.keys().hasAll(['title', 'description', 'options', 'isActive', 'startDate', 'endDate', 'createdBy', 'district', 'category']) &&
             data.title is string && data.title.size() > 0 && data.title.size() <= 200 &&
             data.description is string && data.description.size() > 0 && data.description.size() <= 2000 &&
             data.options is list && data.options.size() >= 2 && data.options.size() <= 10 &&
             data.isActive is bool &&
             data.startDate is timestamp &&
             data.endDate is timestamp &&
             data.createdBy is string &&
             data.district is string &&
             data.category is string;
    }
    
    function isValidOptionIndex(index, maxOptions) {
      return index >= 0 && index < maxOptions;
    }
    
    function isPollActive() {
      let now = request.time;
      return resource.data.startDate <= now && 
             resource.data.endDate >= now &&
             resource.data.isActive == true;
    }
    
    // Polls Collection - Avstemninger
    match /polls/{pollId} {
      // Alle kan lese aktive avstemninger
      allow read: if true;
      
      // Kun admin-brukere kan opprette avstemninger
      allow create: if isAdmin() && 
                       isValidPollData() &&
                       request.resource.data.createdBy == request.auth.uid;
      
      // Kun admin kan oppdatere, og kun med gyldig data
      allow update: if isAdmin() && 
                       isValidPollData();
      
      // Ingen kan slette avstemninger
      allow delete: if false;
      
      // Votes Collection - Stemmer
      match /votes/{voteId} {
        // Kun eier kan lese sin egen stemme
        allow read: if isOwner(resource.data.userId);
        
        // Kun autentiserte brukere med verifisert e-post kan stemme
        // Kun én stemme per bruker per poll
        allow create: if isAuthenticated() && 
                         hasVerifiedEmail() &&
                         request.resource.data.userId == request.auth.uid &&
                         request.resource.data.pollId == pollId &&
                         isValidOptionIndex(request.resource.data.optionIndex, get(/databases/$(database)/documents/polls/$(pollId)).data.options.size()) &&
                         isPollActive() &&
                         !exists(/databases/$(database)/documents/polls/$(pollId)/votes/$(request.auth.uid));
        
        // Ingen kan oppdatere eller slette stemmer
        allow update: if false;
        allow delete: if false;
      }
    }
    
    // Users Collection - Brukerdata
    match /users/{userId} {
      // Brukere kan lese sin egen data
      allow read: if isOwner(userId);
      
      // Brukere kan opprette sin egen profil ved første innlogging
      allow create: if isAuthenticated() && 
                       request.resource.data.uid == request.auth.uid &&
                       request.resource.data.email == request.auth.token.email;
      
      // Brukere kan oppdatere sin egen profil (men ikke UID eller e-post)
      allow update: if isOwner(userId) &&
                       request.resource.data.uid == resource.data.uid &&
                       request.resource.data.email == resource.data.email;
      
      // Ingen kan slette brukerdata
      allow delete: if false;
    }
    
    // News Collection - Nyheter (for fremtidig bruk)
    match /news/{newsId} {
      // Alle kan lese publiserte nyheter
      allow read: if resource.data.publishedAt <= request.time;
      
      // Kun admin kan opprette/oppdatere nyheter
      allow create: if isAdmin();
      allow update: if isAdmin();
      allow delete: if false;
    }
    
    // Rate limiting collection (for server-side rate limiting)
    match /rateLimits/{userId} {
      // Kun server kan lese/skrive
      allow read, write: if false;
    }
    
    // Standard: Ingen tilgang til andre collections
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
